Google Sheets Setup
  1. Go to sheets.google.com → create a new blank sheet → rename the tab data
  2. Tap Extensions → Apps Script (use desktop mode in browser)
  3. Delete existing code and paste the script below → Save
  4. Tap Deploy → New deployment → Web app
  5. Execute as: Me · Who has access: Anyone → Deploy → Authorize
  6. Copy the Web App URL and paste it above
Apps Script Code
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName('data');
  var cell = sh.getRange('A1').getValue();
  return ContentService
    .createTextOutput(cell ? cell : '{}')
    .setMimeType(ContentService.MimeType.JSON);
}
function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName('data');
  sh.getRange('A1').setValue(e.postData.contents);
  return ContentService
    .createTextOutput('ok')
    .setMimeType(ContentService.MimeType.TEXT);
}
